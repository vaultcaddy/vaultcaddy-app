def translate_content(content, translations):
    """批量翻譯150+條目"""
    for english, target_lang in translations.items():
        content = re.sub(
            r'\b' + re.escape(english) + r'\b', 
            target_lang, 
            content, 
            flags=re.IGNORECASE
        )
    return content
